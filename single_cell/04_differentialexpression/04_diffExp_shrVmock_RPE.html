# Create a custom list of clusters to compare
clusters = ['0', '1', '2', '4', '9']

# Initialize an empty list to store dataframes
dfs = []

# Perform differential expression between Base and Null for each cluster
for cluster in clusters:
    # First make a subset of the data
    adata_subset = adata[adata.obs['leiden_r1'] == cluster]
    # Then run the test
    sc.tl.rank_genes_groups(adata_subset, 'group', method='wilcoxon', key_added='wilcoxon', groups=['Null'], reference='Base')
    # Extract the results into a dataframe
    df = sc.get.rank_genes_groups_df(adata_subset, group=None, key='wilcoxon')
    # Add a column to indicate the cluster
    df['group'] = "Cluster" + cluster
    # Append the dataframe to the list
    dfs.append(df)

# Combine all dataframes into a single dataframe
combined_df = pd.concat(dfs, ignore_index=True)

# Place cluster column at the beginning
cols = combined_df.columns.tolist()
cols = cols[-1:] + cols[:-1]
combined_df = combined_df[cols]
